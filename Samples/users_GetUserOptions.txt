  // ----------------------------------------------------  // users_GetUserOptions : Get user login Options  //  // Parameters:  //     $1: username  //  // Return:  //     $0: user options object, with at least the following attributes  //         userID: the user ID in the database  //         isAdmin = true: indicates user has admin privileges  //         isReadONly = true: indicates user has read only access to the database  //  // Assumptions:  //    additional attributes can be added to this object and will be available on the front end side  //    for the web/mobile app peruse  //  //  // ----------------------------------------------------  //C_OBJECT($0;$options)C_TEXT($1;$username)  //--- copy parameters to locals$username:=$1  //--- localsC_LONGINT($i)C_BOOLEAN($isreadonly;$isadmin)  //--- codeGET USER LIST($aUsers;$aIDs)  // get list of Users in the database$i:=Find in array($aUsers;$username)  // find the user on the listIf ($i>0)  // is this a valid user?If (($aIDs{$i}=1) | ($aIDs{$i}=2))  // is user is desigenr or administrator$isadmin:=TrueElse $isadmin:=FalseEnd if   //  // just for illustration, we assume users in a group named READONLY have Read Only access to the databaseIf (User in group($username;"READONLY") & Not($isadmin))$isreadonly:=TrueElse $isreadonly:=FalseEnd if   // set options attributesOB SET($options;"userID";$aIDs{$i};"isReadOnly";$isreadonly;"isAdmin";$isadmin)End if $0:=$options
