  // ----------------------------------------------------  // LISTListToArray : Wrap List To Array function to Ignore disabled items  //  // Parameters:  //     $1: list name  //     $2: array pointer  //  //  // Assumptions:  //  //C_TEXT($1;$listName)C_POINTER($2;$arrayPtr)  //--- copy parameters to locals$listName:=$1$arrayPtr:=$2  //--- locals  //--- codeLIST TO ARRAY($listName;$arrayPtr->)  // set array to # of items in list
